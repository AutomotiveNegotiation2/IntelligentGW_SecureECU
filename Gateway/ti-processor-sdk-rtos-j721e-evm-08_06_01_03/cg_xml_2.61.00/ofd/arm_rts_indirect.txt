#-----------------------------------------------------------------------------
# arm_rts_indirect.txt - Configuration file used with call_graph.pl.  Specific
#    to the ARM compiler.  Use in combination with ti_rts_indirect.txt.
#-----------------------------------------------------------------------------
# For details on syntax, run "perldoc call_graph.pl". 
#-----------------------------------------------------------------------------
# The "EABI" lines are the same as the line above them, but without the
# leading '_'.  This is a hack.  These symbols are defined in C, and thus
# will appear with leading "_" when built with a COFF-based ABI, and
# without the leading "_" when built with EABI.
#-----------------------------------------------------------------------------

# These lines mean these functions make some indirect calls to functions
# that are not named.  In this case, they are the constructors and destructors
# for C++ objects.  Actually listing the names of these functions here would
# be ideal.  But since they are mangled names, it would be difficult.
_c_int00       : 
_call_dtors    :
call_dtors     :                  # EABI
__TI_auto_init :

_memalign      : __nop
memalign       : _nop

# Uncomment the next lines if any global deconstructors are called
#_exit         : _call_dtors
#exit          : call_dtors       # EABI
