#=============================================================================
# General syntax ...
#
#    <section name> <space or ',' or '='> <CODE|DATA|UDATA|IGNORE>
#
# - Each section specification must be contained on a single line
# - Optional space can surround the section name or classification
# - Blank lines are ignored
# - '#' is for comments.  Anything on a line after '#' is ignored.
#=============================================================================
# This particular specification matches the default classification lib_size.pl
# uses if no --sect_classes=file option is used
#=============================================================================
.text           CODE
.bios           CODE
.sysinit        CODE
.hwi_vec        CODE
.rtdx_text      CODE

.data           DATA
.cinit          DATA         # change to IGNORE if link with -cr
.pinit          DATA
.switch         DATA
.const          DATA
.gblinit        DATA
.rtdx_data      DATA
.trcdata        DATA

.bss            UDATA
.far            UDATA
.sysmem         UDATA
.stack          UDATA
.sysstack       UDATA
.args           UDATA
.cio            UDATA
.sysdata        UDATA

# Note no sections are marked IGNORE.  If you link with -cr, you would want
# to change .cinit to IGNORE.  Details are in the comments in lib_size.pl.
